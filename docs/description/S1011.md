# Use a single append to concatenate two slices

Before:

    for _, e := range y {
        x = append(x, e)
    }
    
    for i := range y {
        x = append(x, y[i])
    }
    
    for i := range y {
        v := y[i]
        x = append(x, v)
    }

After:

    x = append(x, y...)
    x = append(x, y...)
    x = append(x, y...)
